MX Script Documentation

A modern open-source scripting language for web apps and APIs. Routes, JSON, JWT, SQL, WebSockets, and AI are all built into the language. Run with mx run app.mx.

Install

Pre-built binaries for macOS / Linux / Windows ship on every tagged release. From source:

git clone https://github.com/jlkdevelop/mxscript.git
cd mxscript
go build -o mx .   # requires Go 1.25+
./mx version

Hello world

$ mx run --eval 'print("hello, " + uuid())'
hello, 4c5e3a17-...

Your first API

server { port: 8080 }

let users = [
  { id: 1, name: "Jassim" },
  { id: 2, name: "Ada" }
]

get /users {
  return json(users)
}

get /users/:id {
  let id = num(request.params.id)
  let user = find(users, fn(u) { return u.id == id })
  if (user == null) {
    return status(404, { error: "not found" })
  }
  return json(user)
}

Save as app.mx. Run mx run app.mx. curl localhost:8080/users/1.

Syntax

Variables, control flow, functions:

let name = "Jassim"
let user = { id: 1, name: name, scores: [10, 20, 30] }

// String interpolation
print("Hello, ${user.name}! Sum: ${sum(user.scores)}")

// Optional chaining + nullish coalescing
let city = user?.profile?.city ?? "unknown"

// Pattern matching
let label = match user.role {
  "admin" => "all access"
  "user"  => "read-only"
  _       => "anonymous"
}

// Loops
loop user.scores as i, score { print(i, score) }
while (count < 100) { count = count + 1 }

// try as expression
let parsed = try { json_parse(input) } catch (e) { { error: e.message } }

Routes

get /             { return text("hello") }
get /users/:id    { return json({ id: request.params.id }) }
post /users       { return status(201, request.body) }
delete /users/:id { return json({ deleted: true }) }

// Verbose form
route GET /admin  { ... }

// Real-time
sse /events       { send({ tick: now() }); sleep(1000) }
ws /chat          { while (true) { send("echo: " + recv()) } }

Middleware

middleware require_auth {
  let claims = jwt.verify(request.bearer_token, env("JWT_SECRET"))
  if (claims == null) {
    return status(401, { error: "unauthorized" })
  }
}

post /admin {
  use require_auth
  return json({ ok: true })
}

Concurrency

let ch = chan(10)
spawn {
  loop 5 as i { send(ch, i * 2) }
  close_chan(ch)
}
while (true) {
  let v = recv(ch)
  if (v == null) { break }
  print(v)
}

// Fan-out / wait-all
let wg = wait_group()
loop urls as u {
  wg.add(1)
  spawn { fetch(u); wg.done() }
}
wg.wait()

HTTP

// Outbound
let res = fetch("https://api.example.com/users", {
  method: "POST",
  headers: { "Content-Type": "application/json" },
  body: { name: "Jassim" }

SQL

let db = sql.open("./data.db")

sql.exec(db, "CREATE TABLE IF NOT EXISTS users (id INTEGER PRIMARY KEY, name TEXT)")
let r = sql.exec(db, "INSERT INTO users (name) VALUES (?)", "Jassim")
print(r.last_insert_id)

let users = sql.query(db, "SELECT * FROM users WHERE name LIKE ?", "Jass%")
loop users as u { print(u.id, u.name) }

sql.close(db)

JWT & OAuth

// JWT
let token = jwt.sign({ sub: "jassim", exp: now() / 1000 + 3600 }, env("JWT_SECRET"))
let claims = jwt.verify(token, env("JWT_SECRET"))   // null on bad sig / expired

// OAuth2 — Google login in two routes
get /auth/google {
  return redirect(oauth.authorize_url({
    provider: "google",
    client_id: env_required("GOOGLE_CLIENT_ID"),
    redirect_uri: "https://app.example.com/auth/callback",
    scopes: ["openid", "email", "profile"]
  }))
}

get /auth/callback {
  let tokens = oauth.exchange_code({
    provider: "google",
    client_id: env_required("GOOGLE_CLIENT_ID"),
    client_secret: env_required("GOOGLE_CLIENT_SECRET"),
    redirect_uri: "https://app.example.com/auth/callback",
    code: request.query.code
  })
  return json(tokens)
}

Real-time

sse /events {
  while (true) {
    send({ time: now_iso() })
    sleep(1000)
  }
}

ws /chat {
  while (true) {
    let msg = recv()
    if (msg == null) { break }
    send("echo: " + msg)
  }
}

CLI

mx run <file.mx>             # run a program
mx run <file.mx> --watch     # hot reload
mx run --eval '<snippet>'    # inline one-liner
mx test [path]               # run *_test.mx files
mx fmt -w <file.mx>          # format in place
mx repl                      # interactive REPL
mx init [name]               # scaffold a new project
mx build <file.mx>           # validate without running
mx build --vercel app.mx     # generate a Vercel-deployable Go project
mx version
mx help
